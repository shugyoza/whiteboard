function twoSum(nums, target) {

  let chk = {};
  for (let i = 0; i < nums.length; i++) {
    let n = nums[i];
    let diff = target - n;
    if (chk[n]) {
      return [chk[n][1], i];
    }
    else chk[diff] = [n, i];
  }
  return [];
}
