function productExceptSelf(nums) {
  const res = [];
  let tmp = 1;
  for (let i = 0; i < nums.length; i++) {
    let a = nums[i];
    res.push(tmp);
    tmp = tmp * a;
  }
  tmp = 1;
  for (let j = nums.length - 1; j >= 0; j--) {
    let b = nums[j];
    res[j] = res[j] * tmp;
    tmp = tmp * b;
  }
  return res;
}
