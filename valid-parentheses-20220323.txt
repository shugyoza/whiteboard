const isValid = (s) => {
  let stk = [];				// [')'
  for (let i = 0; i < s.length; i++) {	// i = 1 < 2 => T
    let c = s[i];			// c = ')'
    if (c === '(') stk.push(')');	// F
    else if (c === '{') stk.push('}');	// F
    else if (c === '[') stk.push(']');	// F
    else {
      let pair = stk.pop();    		// pair = ')'
      if (c !== pair) return false;	// F
    }
  }
  return true;
}
