function maxProfit(prices) {
  let sell = prices[prices.length - 1];
  let profit = 0;
  for (let i = prices.length - 2; i >= 0; i--) {
    let currentPrice = prices[i];
    sell = Math.max(sell, currentPrice);
    profit = Math.max(profit, sell - currentPrice);
  }
  return profit;
}
